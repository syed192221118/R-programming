# 1. Create vectors
numeric_vector <- c(1.5, 2.3, 3.6, 4.8, 5.9)
character_vector <- c("Apple", "Banana", "Cherry")
logical_vector <- c(TRUE, FALSE, TRUE, FALSE)

# Display content and types of vectors
cat("Numeric Vector:\n")
print(numeric_vector)
cat("Type:", class(numeric_vector), "\n\n")

cat("Character Vector:\n")
print(character_vector)
cat("Type:", class(character_vector), "\n\n")

cat("Logical Vector:\n")
print(logical_vector)
cat("Type:", class(logical_vector), "\n\n")
